                function isPrime(num) { 
                    if (num < 2) {
                        return false;
                    }
                    for(let i = 2; i<= Math.sqrt(num); i++) {
                        if (num % i === 0) {
                            return false;
                        }
                    }
                    return true;
                }

  9. Write a function that capitalizes the first letter of each word in a sentence. 
                
                // Alogrithm to Capitalizes the first letter of each sentence
                    /**
                    * Split the sentence into words
                    * iterate through each word
                    * capitalize the first letter of each word by taking the first letter (charAt(0)) making it uppercase and combining it with the rest of the word.
                    * Join the word
                    */
                    function capitalizeFirstLetter(sentence) {
                        let words = sentence.split(" ");
                        for(let i = 0; i< words.length; i++) {
                            let word = words[i];
                            words[i] = word.charAt(0).toUpperCase() + word.slice(1);
                        }
                        sentence = words.join(" ");
                        return sentence;
                    }

                    console.log(capitalizeFirstLetter("hi i am suresh pradhan"));

              // Alogrithm to generate Random Password
              /**
              * 1. Initialize a variable with password with empty value and charset and includes the characters from where you are going to create a password.
              * 2. loop through each character of the charset and create a random Index number
              * 3. Add those randomly generated index number to refer to the position of the character in the charset variable

              function generateRandomPassword(length) {

                for(let i = 0; i< length; i++) {
